 64@app.post("/internal_token")
 65async def internal_token(request: InternalTokenRequest) -> InternalTokenResponse:
 66    """
 67    Get an internal session token via browser authentication.
 68
 69    Checks cache first. If token exists in cache, returns it immediately.
 70    Otherwise, initiates browser authentication flow and caches the result.
 71    """
 72    # Check if we already have a cached token for this domain
 73    if request.domain in _token_cache:
 74        logger.info(f"Using cached token for {request.domain}")
 75        return InternalTokenResponse(token=_token_cache[request.domain], domain=request.domain, cached=True)
 76
 77    # No cached token, get a new one
 78    logger.info(f"No cached token for {request.domain}, initiating authentication")
 79    # Use _skip_server_check=True to prevent recursion (server shouldn't call itself)
 80    # Run in thread pool to avoid blocking event loop and allow signal handling
 81    loop = asyncio.get_event_loop()
 82    with ThreadPoolExecutor() as executor:
 83        token = await loop.run_in_executor(
 84            executor, partial(get_internal_token, request.domain, _skip_server_check=True)
 85        )
 86
 87    # Cache the token for future requests
 88    _token_cache[request.domain] = token
 89    logger.info(f"Cached new token for {request.domain}")
 90
 91    return InternalTokenResponse(token=token, domain=request.domain, cached=False)

168def _prepare_proxy_log_data(request: ProxyRequest, full_url: str, status_code: int) -> dict[str, Any]:
169    """Helper to prepare log data for proxied requests."""
170    from typing import Any as TypingAny
171    from urllib.parse import urlparse
172
173    # Parse URL to extract hostname and path
174    parsed = urlparse(full_url)
175    hostname = parsed.netloc
176    path = parsed.path or "/"
177
178    # Prepare log data
179    log_data: dict[str, TypingAny] = {
180        "method": request.method,
181        "hostname": hostname,
182        "path": path,
183        "url": full_url,
184        "status_code": status_code,
185    }
186
187    # Handle request body - show up to 1028 chars with length indicator if longer
188    body_str: str | None = None
189    if request.json_body is not None:
190        import json as json_module
191
192        body_str = json_module.dumps(request.json_body, ensure_ascii=False)
193    elif request.content is not None:
194        body_str = request.content
195
196    if body_str is not None:
197        if len(body_str) <= 1028:
198            log_data["data"] = body_str
199        else:
200            log_data["data"] = body_str[:1028] + "..."
201            log_data["data_length"] = f"{len(body_str)} characters (showing first 1028)"
202
203    return log_data

206@app.post("/proxy")
207async def proxy_request(request: ProxyRequest) -> ProxyResponse:
208    """
209    Proxy an HTTP request through the stolon server.
210
211    This endpoint:
212    1. Receives request intent from client (method, path, body, params)
213    2. Retrieves/refreshes authentication token
214    3. Makes the actual HTTP call to target domain
215    4. Returns response to client
216
217    Benefits:
218    - Server handles all auth token management
219    - Logging happens automatically server-side
220    """
221    domain = request.domain
222    method = request.method
223    path = request.path
224
225    # Get or refresh token (uses existing cache)
226    token_response = await internal_token(InternalTokenRequest(domain=domain))
227    token = token_response.token
228
229    # Build full URL
230    full_url = f"https://{domain}{path}"
231
232    # Make the actual HTTP call
233    headers = {
234        "Cookie": f"internalSession={token}",
235        "X-Clover-Appenv": f"{request.environment_name}:{domain.split('.')[0]}",
236    }
237
238    # Set Content-Type based on what's provided
239    if request.json_body is not None:
240        headers["Content-Type"] = "application/json"
241    elif request.content_type is not None:
242        headers["Content-Type"] = request.content_type
243    else:
244        headers["Content-Type"] = "application/json"  # Default
245
246    async with httpx.AsyncClient() as client:
247        # Make initial request
248        response = await _make_http_request(client, method, full_url, headers, request)
249
250        # Handle 401 - token expired, refresh and retry
251        if response.status_code == 401:
252            logger.warning(f"Received 401 for {domain}, refreshing token and retrying")
253            # Invalidate cached token
254            if domain in _token_cache:
255                del _token_cache[domain]
256            # Get fresh token
257            token_response = await internal_token(InternalTokenRequest(domain=domain))
258            token = token_response.token
259            headers["Cookie"] = f"internalSession={token}"
260
261            # Retry request
262            response = await _make_http_request(client, method, full_url, headers, request)
263
264        # Log request/response automatically
265        log_data = _prepare_proxy_log_data(request, full_url, response.status_code)
266        logger.info("Proxied HTTP request", **log_data)
267
268        # Return response to client
269        return ProxyResponse(
270            status_code=response.status_code,
271            headers=dict(response.headers),
272            body=response.text,
273        )

335def _deserialize_kwargs(func: object, kwargs: dict[str, object], service: OpenAPIService) -> dict[str, object]:
336    """Deserialize kwargs based on function signature."""
337    import inspect
338
339    from stolon.serialization import deserialize_argument
340
341    # Get function signature
342    sig = inspect.signature(func)  # type: ignore[arg-type]
343
344    deserialized: dict[str, object] = {}
345
346    for param_name, param_value in kwargs.items():
347        if param_name in sig.parameters:
348            param = sig.parameters[param_name]
349            # Get type hint as string
350            if param.annotation != inspect.Parameter.empty:
351                type_hint = str(param.annotation)
352                # Remove module prefixes to get just the class name
353                # (e.g., "stolon.openapi_generated...models.ModelName" -> "ModelName")
354                if "." in type_hint:
355                    parts = type_hint.split(".")
356                    type_hint = parts[-1].strip("'>")  # Handle forward references like "<class 'ModelName'>"
357
358                # Deserialize the parameter
359                deserialized[param_name] = deserialize_argument(param_value, type_hint, service.value)  # type: ignore[arg-type]
360            else:
361                # No type hint, pass as-is
362                deserialized[param_name] = param_value
363        else:
364            # Parameter not in signature, pass as-is
365            deserialized[param_name] = param_value
366
367    return deserialized

370async def _execute_openapi_call(
371    func: object, auth_client: object, kwargs: dict[str, object], request: OpenAPIInvokeRequest
372) -> OpenAPIInvokeResponse:
373    """Execute OpenAPI function call and serialize result."""
374    from stolon.serialization import serialize_result
375
376    try:

378        deserialized_kwargs = _deserialize_kwargs(func, kwargs, request.service)

381
382        if asyncio.iscoroutine(result):  # type: ignore[arg-type]
383            result = await result
384
385        serialized_result = serialize_result(result)  # type: ignore[arg-type]
386        status_code = _extract_status_code(result)  # type: ignore[arg-type]
387
388        logger.info(

391            function_path=request.function_path,
392            status_code=status_code,
393        )
394
395        return OpenAPIInvokeResponse(
396            success=True,
397            result=serialized_result,
398            status_code=status_code,
399        )

404            function_path=request.function_path,
405            error=str(e),
406        )
407        return OpenAPIInvokeResponse(

419@app.post("/invoke_openapi")
420async def invoke_openapi(request: OpenAPIInvokeRequest) -> OpenAPIInvokeResponse:
421    """
422    Invoke an OpenAPI-generated function on the server.
423
424    This endpoint:
425    1. Receives function path and serialized arguments
426    2. Dynamically imports the OpenAPI function
427    3. Creates an authenticated client with cached token
428    4. Calls the function and handles auth retries
429    5. Returns serialized result
430
431    Benefits:
432    - Server executes the full OpenAPI client (no parsing bugs)
433    - Auth token management stays on server
434    - Client code is simpler (just serialize/deserialize)
435    """
436    try:
437        # Import the OpenAPI function
438        logger.info(
439            "Invoking OpenAPI function",
440            service=request.service,
441            function_path=request.function_path,
442            variant=request.variant,
443        )
444
445        try:
446            func = _import_openapi_function(request.service, request.function_path, request.variant)
447        except (ImportError, AttributeError) as e:
448            logger.error("Failed to import OpenAPI function", error=str(e))
449            return OpenAPIInvokeResponse(success=False, error=f"Failed to import function: {e}")
450
451        # Get token and create client
452        token_response = await internal_token(InternalTokenRequest(domain=request.domain))
453        token = token_response.token
454
455        base_path = _get_service_base_path(request.service)
456        auth_client = _create_authenticated_client(
457            request.service, request.domain, token, request.environment_name, base_path
458        )
459
460        # Call the function
461        response = await _execute_openapi_call(func, auth_client, request.kwargs, request)
462
463        # Check if we got a 401 (either as exception or status code)
464        needs_retry = False
465        if (
466            not response.success
467            and response.error
468            and _is_auth_error(Exception(response.error))
469            or response.success
470            and response.status_code == 401
471        ):
472            needs_retry = True
473
474        if needs_retry:
475            logger.warning(
476                f"Received 401 for {request.domain}, refreshing token and retrying",
477                service=request.service,
478            )
479
480            # Invalidate cached token
481            if request.domain in _token_cache:
482                del _token_cache[request.domain]
483
484            # Get fresh token and create new client
485            token_response = await internal_token(InternalTokenRequest(domain=request.domain))
486            token = token_response.token
487            auth_client = _create_authenticated_client(
488                request.service, request.domain, token, request.environment_name, base_path
489            )
490
491            # Retry the function call
492            response = await _execute_openapi_call(func, auth_client, request.kwargs, request)
493
494            if response.success:
495                logger.info(
496                    "OpenAPI function succeeded after retry",
497                    service=request.service,
498                    function_path=request.function_path,
499                    status_code=response.status_code,
500                )
501
502        return response
503
504    except Exception as e:
505        logger.error("invoke_openapi failed", error=str(e))
506        return OpenAPIInvokeResponse(
507            success=False,
508            error=f"Server error: {e}",
509        )
