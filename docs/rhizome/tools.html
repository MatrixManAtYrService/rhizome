rhizome.tools

Tool abstractions for external command dependencies.

This module provides injectable tool dependencies to enable testing and mocking of external tools like kubectl, gcloud, 1password CLI, and lsof.

132class ExternalKubectlTool(KubectlTool):
133    """External kubectl tool using subprocess."""
134
135    async def cluster_info(self, context: str) -> CommandResult:
136        """Get cluster information."""
137        return await self._run_command(["kubectl", "cluster-info", "--context", context])
138
139    async def exec_in_pod(self, context: str, namespace: str, deployment: str, command: list[str]) -> CommandResult:
140        """Execute command in a pod."""
141        args = [
142            "kubectl",
143            "--context",
144            context,
145            "-n",
146            namespace,
147            "exec",
148            "-it",
149            f"deployment/{deployment}",
150            "--",
151        ] + command
152        return await self._run_command(args)
153
154    async def get_logs(self, context: str, namespace: str, deployment: str, since: str = "10s") -> list[LogLine]:
155        """Get recent logs from a deployment."""
156        result = await self._run_command(
157            ["kubectl", "--context", context, "-n", namespace, "logs", f"deployment/{deployment}", "--since", since]
158        )
159
160        if not result.success:
161            return []
162
163        return [LogLine(content=line) for line in result.stdout.splitlines() if line.strip()]
164
165    async def port_forward(
166        self, context: str, namespace: str, resource: str, local_port: int, remote_port: int
167    ) -> asyncio.subprocess.Process:
168        """Start port forwarding process (returns running process)."""
169        args = [
170            "kubectl",
171            "--context",
172            context,
173            "-n",
174            namespace,
175            "port-forward",
176            resource,
177            f"{local_port}:{remote_port}",
178        ]
179
180        process = await asyncio.create_subprocess_exec(*args)
181        return process
182
183    async def get_pod_status(self, pod_name: str, context: str | None = None) -> CommandResult:
184        """Get pod status."""
185        args = ["kubectl", "get", "pod", pod_name, "-o", "jsonpath={.status.phase}"]
186        if context:
187            args.extend(["--context", context])
188        return await self._run_command(args)
189
190    async def unset_cluster_ca(self, cluster_name: str) -> CommandResult:
191        """Unset cluster certificate authority data."""
192        return await self._run_command(
193            ["kubectl", "config", "unset", f"clusters.{cluster_name}.certificate-authority-data"]
194        )
195
196    async def _run_command(self, args: list[str]) -> CommandResult:
197        """Run a kubectl command and return result."""
198        process = await asyncio.create_subprocess_exec(
199            *args, stdout=asyncio.subprocess.PIPE, stderr=asyncio.subprocess.PIPE
200        )
201
202        stdout, stderr = await process.communicate()
203
204        return CommandResult(
205            returncode=process.returncode or 0,
206            stdout=stdout.decode() if stdout else "",
207            stderr=stderr.decode() if stderr else "",
208        )

256class ExternalLsofTool(LsofTool):
257    """External lsof tool using subprocess."""
258
259    async def check_port(self, port: int) -> list[PortInfo]:
260        """Check if a port is in use."""
261        try:
262            process = await asyncio.create_subprocess_exec(
263                "lsof", "-i", f":{port}", stdout=asyncio.subprocess.PIPE, stderr=asyncio.subprocess.PIPE
264            )
265
266            stdout, _ = await process.communicate()
267
268            if process.returncode != 0:
269                return []  # Port not in use
270
271            lines = stdout.decode().splitlines()
272            port_info: list[PortInfo] = []
273
274            for line in lines[1:]:  # Skip header
275                parts = line.split()
276                if len(parts) >= 2:
277                    port_info.append(
278                        PortInfo(pid=int(parts[1]) if parts[1].isdigit() else 0, process_name=parts[0], port=port)
279                    )
280
281            return port_info
282
283        except Exception:
284            return []  # Error checking port, assume not in use

287class ExternalPybritiveTool(PybritiveTool):
288    """External pybritive tool using subprocess."""
289
290    DEFAULT_PATTERN = r"""
291        Temp\sMySQL\susername:\s*(?P<username>\S+).*
292        Temp\spassword:\s*(?P<password>\S+).*
293        For\sbilling\sin\susprod\sconnect\sto\sserver:\s*(?P<host>[^:]+):(?P<port>\d+)
294    """
295
296    async def checkout(self, resource_path: str, pattern: str | None = None) -> BritiveInfo:
297        """Checkout temporary credentials from Britive."""
298        import re
299
300        import structlog
301
302        log = structlog.get_logger()
303        result = await self._run_command(["pybritive", "checkout", resource_path])
304
305        if not result.success:
306            log.error("Pybritive checkout failed", stderr=result.stderr, stdout=result.stdout)
307            raise RuntimeError(f"Pybritive checkout failed: {result.stderr}")
308
309        log.debug("Pybritive raw output", output=result.stdout)
310
311        # Use the provided pattern or the default
312        checkout_pattern = pattern or self.DEFAULT_PATTERN
313        match = re.search(checkout_pattern, result.stdout, re.DOTALL | re.VERBOSE)
314
315        if not match:
316            log.error(
317                "Failed to parse pybritive output with regex",
318                pattern=checkout_pattern,
319                output=result.stdout,
320            )
321            raise RuntimeError(f"Failed to parse pybritive output: {result.stdout}")
322
323        credentials = match.groupdict()
324        username = credentials.get("username", "")
325        password = credentials.get("password", "")
326        host = credentials.get("host", "")
327        port = int(credentials.get("port", 0))
328
329        # Log parsed credentials (redacting password)
330        log.info(
331            "Pybritive credentials parsed",
332            username=username,
333            password="[REDACTED]" if password else "",
334            host=host,
335            port=port,
336            resource_path=resource_path,
337        )
338
339        if not all([username, password, host, port]):
340            log.error(
341                "Failed to extract all credentials from pybritive output",
342                missing_fields={
343                    "username": bool(username),
344                    "password": bool(password),
345                    "host": bool(host),
346                    "port": bool(port),
347                },
348            )
349            raise RuntimeError(f"Failed to extract all credentials from pybritive output: {result.stdout}")
350
351        return BritiveInfo(username=username, password=password, host=host, port=port)
352
353    async def _run_command(self, args: list[str]) -> CommandResult:
354        """Run a pybritive command and return result."""
355        import asyncio
356
357        process = await asyncio.create_subprocess_exec(
358            *args, stdout=asyncio.subprocess.PIPE, stderr=asyncio.subprocess.PIPE
359        )
360
361        stdout, stderr = await process.communicate()
362
363        return CommandResult(
364            returncode=process.returncode or 0,
365            stdout=stdout.decode() if stdout else "",
366            stderr=stderr.decode() if stderr else "",
367        )

370class Tools:
371    """Container for all external tool dependencies with dependency injection."""
372
373    def __init__(
374        self,
375        kubectl: KubectlTool | None = None,
376        gcloud: GcloudTool | None = None,
377        onepassword: OnePasswordTool | None = None,
378        lsof: LsofTool | None = None,
379        pybritive: PybritiveTool | None = None,
380    ) -> None:
381        """
382        Initialize tools with optional dependency injection.
383
384        Args:
385            kubectl: kubectl tool (defaults to ExternalKubectlTool)
386            gcloud: gcloud tool (defaults to ExternalGcloudTool)
387            onepassword: 1Password CLI tool (defaults to ExternalOnePasswordTool)
388            lsof: lsof tool (defaults to ExternalLsofTool)
389            pybritive: pybritive tool (defaults to ExternalPybritiveTool)
390        """
391        self.kubectl = kubectl or ExternalKubectlTool()
392        self.gcloud = gcloud or ExternalGcloudTool()
393        self.onepassword = onepassword or ExternalOnePasswordTool()
394        self.lsof = lsof or ExternalLsofTool()
395        self.pybritive = pybritive or ExternalPybritiveTool()
396
397    def is_mocked(self) -> bool:
398        """Check if any of the tools are mocked (non-External implementations)."""
399        return (
400            not self.kubectl.__class__.__name__.startswith("External")
401            or not self.gcloud.__class__.__name__.startswith("External")
402            or not self.onepassword.__class__.__name__.startswith("External")
403            or not self.lsof.__class__.__name__.startswith("External")
404            or not self.pybritive.__class__.__name__.startswith("External")
405        )
