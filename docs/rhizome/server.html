126async def localk8s_credentials_generator(request: LocalK8sRequest) -> AsyncGenerator[str, None]:
127    """
128    Generate SSE stream for local K8s credentials.
129
130    This simulates the credential retrieval process that would normally
131    involve vault, auth flows, etc. For testing, it uses hard-coded credentials
132    with a configurable delay.
133    """
134    # First, start the port forward process
135    logger.info(f"Starting port forward for {request.kube_deployment}")
136
137    try:
138        # Start port forward (but with a fake sql_connection for local k8s)
139        port_forward_response = await start_portforward(
140            kube_context=request.kube_context,
141            kube_namespace=request.kube_namespace,
142            kube_deployment=request.kube_deployment,
143            sql_connection=f"localk8s:{request.kube_namespace}:{request.kube_deployment}",
144            local_port=request.local_port,
145        )
146        logger.info(f"Port forward started with PID {port_forward_response.pid}")
147    except Exception as e:
148        logger.error(f"Failed to start port forward: {e}")
149        yield f"event: error\ndata: {json.dumps({'error': f'Failed to start port forward: {e}'})}\n\n"
150        return
151
152    # Send initial status
153    yield f"event: status\ndata: {json.dumps({'status': 'port_forward_started', 'pid': port_forward_response.pid})}\n\n"
154
155    # Give kubectl a moment to establish the port forward
156    await asyncio.sleep(1.0)
157
158    # Wait for the specified delay (simulating credential retrieval)
159    logger.info(f"Waiting {request.delay}s for credentials...")
160    yield f"event: status\ndata: {json.dumps({'status': 'retrieving_credentials', 'delay': request.delay})}\n\n"
161
162    await asyncio.sleep(request.delay)
163
164    # Send credentials (hard-coded for local testing)
165    credentials = {
166        "username": "user",
167        "password": "pass",
168        "database": "test",
169        "connection_string": f"mysql+pymysql://user:pass@localhost:{request.local_port}/test",
170    }
171
172    logger.info("Credentials retrieved, sending to client")
173    yield f"event: credentials\ndata: {json.dumps(credentials)}\n\n"

195def _get_environment_instance(environment_name: str) -> Any:  # noqa: ANN401
196    """
197    Get environment instance from environment name.
198
199    Args:
200        environment_name: Environment name (e.g., "DevMeta")
201
202    Returns:
203        Environment instance
204
205    Raises:
206        ValueError: If environment not found
207    """
208    from rhizome.client import RhizomeClient
209    from rhizome.environments.environment_list import get_environment_enum_by_name
210
211    env_enum = get_environment_enum_by_name(environment_name)
212    if env_enum is None:
213        raise ValueError(f"Unknown environment: {environment_name}")
214
215    # Get the environment class
216    from rhizome.environments.environment_list import environment_type
217
218    env_class = environment_type.get(env_enum)
219    if not env_class:
220        raise ValueError(f"Unknown environment: {environment_name}")
221
222    # Create environment instance (needs a dummy client)
223    return env_class(RhizomeClient(data_in_logs=False))

226@app.post("/write_query")
227def write_query(request: WriteQueryRequest) -> WriteQueryResponse:
228    """
229    Execute a canned write query with user approval.
230
231    This endpoint:
232    1. Looks up the pre-defined query
233    2. Renders it with parameters
234    3. Prompts the user for approval (yes/no)
235    4. If approved, executes with RW credentials
236    5. Returns the result
237
238    Args:
239        request: Query name, environment, and parameters
240
241    Returns:
242        WriteQueryResponse with execution results
243    """
244    from urllib.parse import quote_plus
245
246    import sqlalchemy
247    from rich.console import Console
248    from rich.panel import Panel
249    from rich.prompt import Confirm
250    from sqlmodel import create_engine
251
252    from rhizome.canned_queries import get_query, render_query
253
254    console = Console()
255
256    try:
257        # Get the canned query
258        query = get_query(request.query_name)
259
260        # Get environment instance
261        env = _get_environment_instance(request.environment_name)
262
263        # Get RW database config
264        db_config_rw = env.get_database_config_rw()
265        if not db_config_rw:
266            return WriteQueryResponse(
267                approved=False,
268                executed=False,
269                error=f"Environment {request.environment_name} does not support write operations",
270            )
271
272        # Render query for display
273        rendered = render_query(request.query_name, request.params)
274
275        # Build the panel content
276        panel_content = (
277            f"[bold cyan]Query:[/bold cyan] {query.name}\n"
278            f"[bold cyan]Description:[/bold cyan] {query.description}\n"
279            f"[bold cyan]Environment:[/bold cyan] {request.environment_name}\n"
280            f"[bold cyan]Database:[/bold cyan] {db_config_rw.database}\n"
281        )
282
283        # Add reason if provided
284        if request.reason:
285            panel_content += f"\n[bold green]Why:[/bold green] {request.reason}\n"
286
287        # Add entity descriptions if provided
288        if request.entity_descriptions:
289            panel_content += "\n[bold magenta]Entities:[/bold magenta]\n"
290            for entity_key, description in request.entity_descriptions.items():
291                panel_content += f"  • [magenta]{entity_key}:[/magenta] {description}\n"
292
293        # Add the SQL query
294        panel_content += f"\n[yellow]{rendered}[/yellow]"
295
296        # Show query to user with rich formatting
297        console.print()
298        console.print(
299            Panel.fit(
300                panel_content,
301                title="[bold red]⚠️  WRITE OPERATION[/bold red]",
302                border_style="red",
303            )
304        )
305        console.print()
306
307        # Prompt user
308        approved = Confirm.ask("[bold]Execute this query?[/bold]", default=False)
309
310        if not approved:
311            console.print("[yellow]Query execution cancelled by user[/yellow]")
312            return WriteQueryResponse(approved=False, executed=False)
313
314        # Execute with RW credentials
315        console.print("[green]Executing query with RW credentials...[/green]")
316
317        encoded_password = quote_plus(db_config_rw.rw_password)
318        connection_string = f"mysql+pymysql://{db_config_rw.rw_username}:{encoded_password}@{db_config_rw.host}:{db_config_rw.port}/{db_config_rw.database}"
319
320        engine = create_engine(connection_string)
321        # For raw SQL with parameters, use engine connection directly
322        with engine.connect() as connection:
323            result = connection.execute(sqlalchemy.text(query.sql), request.params)
324            connection.commit()
325            rows_affected = result.rowcount
326
327        console.print(f"[green]✓ Query executed successfully. Rows affected: {rows_affected}[/green]")
328        console.print()
329
330        return WriteQueryResponse(approved=True, executed=True, rows_affected=rows_affected)
331
332    except Exception as e:
333        error_msg = str(e)
334        console.print(f"[red]✗ Error: {error_msg}[/red]")
335        console.print()
336        return WriteQueryResponse(approved=False, executed=False, error=error_msg)

339def _execute_query_first(
340    session: Any,  # noqa: ANN401
341    request: ExecuteQueryRequest,
342    model_class: type[Any],  # noqa: ANN401
343) -> ExecuteQueryResponse:
344    """Execute query and return first result or None."""
345    import sqlalchemy
346
347    from rhizome.serialization import deserialize_result, serialize_result
348
349    # Use execute() for raw SQL (exec() is for ORM queries)
350    result_proxy = session.execute(sqlalchemy.text(request.sql), request.parameters)  # type: ignore[call-overload]
351    row = result_proxy.first()
352
353    if row is None:
354        return ExecuteQueryResponse(success=True, result=None, row_count=0)
355
356    # Convert row to dict then to model instance
357    # Type ignore: row._mapping is a private SQLAlchemy API with incomplete typing
358    row_dict: dict[str, Any] = dict(row._mapping)  # type: ignore[attr-defined, var-annotated]
359    result = deserialize_result(row_dict, model_class)  # type: ignore[arg-type]
360
361    # Serialize result (handles sanitization)
362    serialized = serialize_result(result, sanitize=request.sanitize)
363    return ExecuteQueryResponse(success=True, result=serialized, row_count=1)

366def _execute_query_all(
367    session: Any,  # noqa: ANN401
368    request: ExecuteQueryRequest,
369    model_class: type[Any],  # noqa: ANN401
370) -> ExecuteQueryResponse:
371    """Execute query and return all results."""
372    import sqlalchemy
373    from sqlmodel import SQLModel
374
375    from rhizome.serialization import deserialize_result, serialize_result_list
376
377    # Use execute() for raw SQL (exec() is for ORM queries)
378    result_proxy = session.execute(sqlalchemy.text(request.sql), request.parameters)  # type: ignore[call-overload]
379    rows = result_proxy.fetchall()
380
381    if not rows:
382        return ExecuteQueryResponse(success=True, result=[], row_count=0)
383
384    # Convert rows to model instances
385    model_results: list[SQLModel] = []
386    for row in rows:
387        # Type ignore: row._mapping is a private SQLAlchemy API with incomplete typing
388        row_dict: dict[str, Any] = dict(row._mapping)  # type: ignore[attr-defined, var-annotated]
389        deserialized = deserialize_result(row_dict, model_class)  # type: ignore[arg-type]
390        if deserialized is not None:
391            model_results.append(deserialized)
392
393    # Serialize results
394    serialized = serialize_result_list(model_results, sanitize=request.sanitize)
395    return ExecuteQueryResponse(success=True, result=serialized, row_count=len(serialized))

400    request: ExecuteQueryRequest,
401    model_class: type[Any],  # noqa: ANN401

428    result = deserialize_result(row_dict, model_class)  # type: ignore[arg-type]
429
430    # Serialize result
431    serialized = serialize_result(result, sanitize=request.sanitize)

435@app.post("/execute_query")
436def execute_query(request: ExecuteQueryRequest) -> ExecuteQueryResponse:
437    """
438    Execute a query server-side and return serialized results.
439
440    This endpoint:
441    1. Gets the environment and database config
442    2. Creates a database connection
443    3. Executes the SQL query with parameters
444    4. Deserializes results into model instances
445    5. Applies sanitization if requested
446    6. Serializes results for transmission
447
448    Args:
449        request: Query execution request with SQL, parameters, model info
450
451    Returns:
452        ExecuteQueryResponse with serialized results or error
453    """
454    from urllib.parse import quote_plus
455
456    from sqlmodel import Session, create_engine
457
458    from rhizome.serialization import import_model_class
459
460    try:
461        # Get environment instance
462        env = _get_environment_instance(request.environment_name)
463
464        # Get database config (read-only)
465        db_config = env.get_database_config()
466
467        # Build connection string
468        encoded_password = quote_plus(db_config.password)
469        connection_string = (
470            f"mysql+pymysql://{db_config.username}:{encoded_password}"
471            f"@{db_config.host}:{db_config.port}/{db_config.database}"
472        )
473
474        # Import the model class for result deserialization
475        model_class = import_model_class(request.model_module, request.model_class)
476
477        # Create engine and execute query
478        engine = create_engine(connection_string)
479
480        with Session(engine) as session:
481            if request.mode == GetMode.FIRST:
482                return _execute_query_first(session, request, model_class)
483            elif request.mode == GetMode.ALL:
484                return _execute_query_all(session, request, model_class)
485            elif request.mode == GetMode.ONE:
486                return _execute_query_one(session, request, model_class)
487            else:
488                return ExecuteQueryResponse(
489                    success=False,
490                    result=None,
491                    error=f"Unknown query mode: {request.mode}",
492                )
493
494    except Exception as e:
495        error_msg = f"{type(e).__name__}: {e}"
496        logger.error("Query execution failed", error=error_msg, environment=request.environment_name)
497        return ExecuteQueryResponse(success=False, result=None, error=error_msg)

502    """
503    Log SQL query details before execution.
504
505    Logs query_id, database, and parameters (if present) via structlog.
506    Prints the statement to stderr so newlines render properly for readability.
507    Does not log result data.
508    """

510    from typing import Any
511
512    # Prepare log data (without statement - we'll print that separately)

515        "database": query.database,
516        "connection_string": query.connection_string,

518
519    # Include parameters if present
520    if query.parameters is not None:
521        log_data["parameters"] = query.parameters
522
523    # Log metadata via structlog
524    logger.info("SQL query", **log_data)
525
526    # Print the statement to stderr with proper newline rendering and indentation
527    import textwrap
528
529    indented_statement = textwrap.indent(query.statement, "    ")
530    print(f"  Statement:\n{indented_statement}", file=sys.stderr)
531
532    return {"status": "logged"}
